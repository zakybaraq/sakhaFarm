    ```typescript
    import React from 'react'
    import ReactDOM from 'react-dom/client'
    import { ThemeProvider } from '@mui/material/styles'
    import CssBaseline from '@mui/material/CssBaseline'
    import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
    import { BrowserRouter } from 'react-router-dom'
    import App from './App'
    import { theme } from './theme'

    const queryClient = new QueryClient({
      defaultOptions: {
        queries: {
          staleTime: 1000 * 60 * 5, // 5 minutes
          retry: 1,
        },
      },
    })

    ReactDOM.createRoot(document.getElementById('root')!).render(
      <React.StrictMode>
        <QueryClientProvider client={queryClient}>
          <ThemeProvider theme={theme}>
            <CssBaseline />
            <BrowserRouter>
              <App />
            </BrowserRouter>
          </ThemeProvider>
        </QueryClientProvider>
      </React.StrictMode>,
    )
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → Server API | Vite proxy forwards /api to localhost:3000 — no credentials exposed in client code |
| Environment → Client | VITE_ prefixed env vars only — prevents server secrets from leaking to browser bundle |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-12 | Information Disclosure | Vite env vars | mitigate | Only VITE_ prefixed variables are bundled into client — server secrets (DATABASE_URL, JWT_SECRET) cannot leak. Enforced by Vite's env prefix mechanism |
| T-01-13 | Spoofing | Vite proxy config | mitigate | Proxy only forwards /api to localhost:3000 — no arbitrary URL proxying. Prevents open proxy vulnerability. CORS handled server-side (Phase 10) |
</threat_model>
