   ```tsx
   useEffect(() => {
     if (editId && data?.users) {
       const user = data.users.find(u => u.id === editId)
       if (user) {
         reset({
           name: user.name,
           email: user.email,
           password: '', // Don't pre-fill password
           roleId: user.roleId,
           tenantId: user.tenantId,
         })
       }
     } else {
       reset({
         email: '',
         password: '',
         name: '',
         roleId: 0,
         tenantId: 0,
       })
     }
   }, [editId, data, reset])
   ```
