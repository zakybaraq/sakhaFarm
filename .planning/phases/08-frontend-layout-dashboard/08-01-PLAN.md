```typescript
export async function getUserProfile(userId: string) {
  const result = await db
    .select({
      id: users.id, email: users.email, name: users.name,
      roleId: users.roleId, tenantId: users.tenantId,
      isActive: users.isActive, isLocked: users.isLocked,
      forcePasswordChange: users.forcePasswordChange,
      lastLoginAt: users.lastLoginAt, createdAt: users.createdAt,
      roleName: roles.name, tenantName: tenants.name,
    })

    .leftJoin(roles, eq(users.roleId, roles.id))
    .leftJoin(tenants, eq(users.tenantId, tenants.id))
```

2. Create `client/src/api/client.ts` — typed fetch wrapper:

CRITICAL BEHAVIOR: The server's `/api/auth/me` endpoint returns HTTP 200 with `{ error: 'Not authenticated' }` or `{ error: 'Session expired' }` for unauthenticated sessions. It does NOT return 401. The client MUST check for the `error` field in the response body, not just HTTP status.

- `API_BASE` constant: `import.meta.env.VITE_API_URL || '/api'` — use RELATIVE path `/api` so Vite proxy handles cookies in dev per D-05 research finding #6
- `ApiError` class extending Error with `status` and `code` fields
- `apiClient<T>(endpoint, options)` function that:
  - Prepends `API_BASE` to endpoint
  - Sets `credentials: 'include'` on ALL requests (per D-03 cookie-based auth)
  - Sets `Content-Type: application/json` header
  - After response comes back: if HTTP status is 401, redirect to `/login` via `window.location.href = '/login'`
  - If HTTP status is not 2xx, parse error body and throw `ApiError`
  - If HTTP status IS 2xx, parse JSON and CHECK for `error` field — if `error` field exists, treat as auth failure:
    - If `error === 'Not authenticated'` or `error === 'Session expired'`, clear auth cache and redirect to `/login`
    - Otherwise throw `ApiError` with the error message
  - If no `error` field, return parsed JSON as typed `T`
  - Export `apiClient` and `ApiError`

2. Add a NEW GET `/api/auth/permissions` endpoint that returns the current user's permissions WITHOUT requiring `rbac.read` permission.

CRITICAL: This solves research finding #2 — regular users need their own permissions for menu filtering, but `/api/rbac/permissions` requires `rbac.read`. This new endpoint uses the authenticated user's `roleId` from the session to call `getRolePermissions(roleId)` directly.

The endpoint handler:
```typescript
.get('/auth/permissions', async ({ cookie }) => {
    const sessionId = cookie.auth_session?.value as string | undefined
    if (!sessionId) {
      return { error: 'Not authenticated' }
    }
    const { user, session } = await validateUserSession(sessionId)
    if (!user || !session) {
      return { error: 'Session expired' }
    }
    const permissions = await getRolePermissions(user.roleId)
    return { permissions }
  })
```

This endpoint must be placed BEFORE the rate-limited `/login` route chain (the `.use(loginRateLimit)` middleware) since it's a read-only GET endpoint that should not count against login rate limits. Place it after the existing `/me` endpoint and before the `/profile` endpoint.

1. Create `client/src/contexts/AuthContext.tsx`:

CRITICAL PROVIDER ORDER (per D-04 research): AuthProvider must be INSIDE QueryClientProvider because it uses TanStack Query hooks. The final order in main.tsx will be: StrictMode > BrowserRouter > QueryClientProvider > ThemeProvider > CssBaseline > AuthProvider > App.

The AuthContext implementation:

```typescript
import { createContext, useContext, useState, useCallback, useEffect } from 'react'
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { login as apiLogin, logout as apiLogout, getCurrentUser } from '../api/auth'
import { getUserPermissions } from '../api/rbac'
import type { User, Permission } from '../types'

interface AuthContextType {
  user: User | null
  isAuthenticated: boolean
  isLoading: boolean
  permissions: string[]
  login: (email: string, password: string) => Promise<void>
  logout: () => Promise<void>
}

const AuthContext = createContext<AuthContextType | null>(null)

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const queryClient = useQueryClient()
  const [permissions, setPermissions] = useState<string[]>([])

  // Check session on mount — CRITICAL: retry: false prevents infinite loops on 401/unauthenticated
  const { data: meData, isLoading } = useQuery({
    queryKey: ['auth', 'me'],
    queryFn: getCurrentUser,
    retry: false,
    staleTime: 1000 * 60 * 5, // 5 minutes
  })

  const user = meData?.user ?? null
  const isAuthenticated = !!meData?.user

  // Fetch permissions when user is authenticated
  useEffect(() => {
    if (isAuthenticated && user) {
      getUserPermissions()
        .then((data) => {
          setPermissions(data.permissions.map((p: Permission) => p.permissionName))
        })
        .catch(() => {
          setPermissions([])
        })
    } else {
      setPermissions([])
    }
  }, [isAuthenticated, user?.id])

  const loginMutation = useMutation({
    mutationFn: ({ email, password }: { email: string; password: string }) =>
      apiLogin(email, password),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['auth'] })
    },
  })

  const logoutMutation = useMutation({
    mutationFn: apiLogout,
    onSuccess: () => {
      queryClient.clear()
      setPermissions([])
    },
  })

  const login = useCallback(
    async (email: string, password: string) => {
      await loginMutation.mutateAsync({ email, password })
    },
    [loginMutation]
  )

  const logout = useCallback(async () => {
    await logoutMutation.mutateAsync()
  }, [logoutMutation])

  return (
    <AuthContext.Provider
      value={{
        user,
        isAuthenticated,
        isLoading,
        permissions,
        login,
        logout,
      }}
    >
      {children}
    </AuthContext.Provider>
  )
}

export function useAuth(): AuthContextType {
  const ctx = useContext(AuthContext)
  if (!ctx) throw new Error('useAuth must be used within AuthProvider')
  return ctx
}
```

NOTE: The `useEffect` for permissions fetch uses `user?.id` as dependency to re-fetch only when the user changes, preventing unnecessary API calls.

```typescript
import { Navigate, useLocation } from 'react-router-dom'
import { CircularProgress, Box } from '@mui/material'
import { useAuth } from '../contexts/AuthContext'

interface RequireAuthProps {
  children: React.ReactNode
  requiredPermission?: string
}

export function RequireAuth({ children, requiredPermission }: RequireAuthProps) {
  const { isAuthenticated, isLoading, permissions } = useAuth()
  const location = useLocation()

  if (isLoading) {
    return (
      <Box sx={{ display: 'flex', justifyContent: 'center', alignItems: 'center', minHeight: '100vh' }}>
        <CircularProgress />
      </Box>
    )
  }

  if (!isAuthenticated) {
    return <Navigate to={`/login?returnUrl=${encodeURIComponent(location.pathname)}`} replace />
  }

  if (requiredPermission && !permissions.includes(requiredPermission)) {
    return <Navigate to="/" replace />
  }

  return <>{children}</>
}
```

The `requiredPermission` prop is optional. Phase 9 will use it for permission-based route guards. For now, RequireAuth only checks authentication.
