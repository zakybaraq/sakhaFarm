```tsx
function PharmaceuticalUsageSection({ cycleId }: { cycleId: number }) {
  const { user } = useAuth()
  const queryClient = useQueryClient()

  // Data: plasmas, pharma items, batches
  const { data: pharmaData } = useQuery({
    queryKey: ['pharmaceuticals', user?.tenantId],
    queryFn: () => listPharmaceuticals({ tenantId: user!.tenantId, isActive: true }),
  })

  const selectedItemId = useWatch({ name: 'itemId' })
  const { data: batchesData } = useQuery({
    queryKey: ['pharma-batches', selectedItemId],
    queryFn: () => selectedItemId ? listPharmaceuticalBatches({ itemId: selectedItemId }) : null,
    enabled: !!selectedItemId,
  })

  const form = useForm<UsageForm>({ /* zod schema from RESEARCH */ })

  const mutation = useMutation({
    mutationFn: recordPharmaceuticalUsage,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['pharmaceutical-stock'] })
      showSuccessSnackbar()
    }
  })

  return (
    <Paper sx={{ p: 3, mt: 4 }}>
      <Typography variant="h6">Penggunaan Obat/Vitamin</Typography>

      {/* Form fields: date, plasma, item, batch+quantity rows, note */}
      <Box sx={{ display: 'flex', flexDirection: 'column', gap: 2 }}>
        {/* Date field */}
        <Controller name="date" control={control} render={({ field }) => (
          <TextField {...field} label="Tanggal" type="date" InputLabelProps={{ shrink: true }} />
        )} />

        {/* Plasma autocomplete */}
        <Controller name="plasmaId" control={control} render={({ field }) => (
          <Autocomplete options={plasmas} getOptionLabel={(p) => p.name} ... />
        )} />

        {/* Item autocomplete */}
        <Controller name="itemId" control={control} render={({ field }) => (
          <Autocomplete
            options={pharmaData?.items ?? []}
            getOptionLabel={(i) => `${i.code} - ${i.name}`}
            ...
          />
        )} />

        {/* Batch + quantity rows (dynamic) */}
        {selectedItemId && batchesData?.batches && (
          <Box>
            {form.getValues('batchQuantities').map((bq, idx) => (
              <Box key={idx} sx={{ display: 'flex', gap: 1 }}>
                <Controller name={`batchQuantities.${idx}.batchId`} ... />
                <Controller name={`batchQuantities.${idx}.quantity`} ... />
              </Box>
            ))}
          </Box>
        )}

        {/* Note */}
        <Controller name="note" ... />

        {/* Submit */}
        <Button onClick={form.handleSubmit((data) => mutation.mutate({
          cycleId,
          ...data,
        }))}>
          Simpan Penggunaan
        </Button>
      </Box>
    </Paper>
  )
}
```
